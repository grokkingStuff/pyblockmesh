* Example


#+BEGIN_SRC python
import pyblockmesh as pbm
v0 = pbm.Vertex(-1,-1,-1)
v1 = pbm.Vertex( 1,-1,-1)
v2 = pbm.Vertex( 1, 1,-1)
v3 = pbm.Vertex(-1, 1,-1)
v4 = pbm.Vertex(-1,-1, 1)
v5 = pbm.Vertex( 1,-1, 1)
v6 = pbm.Vertex( 1, 1, 1)
v6 = pbm.Vertex(-1, 1, 1)

h1 = pbm.Hexahedron([v0,v1,v2,v3,v4,v5,v6,v7])
